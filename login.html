""" Models for the messaging application. This module contains the database models for the WhatsApp-like messaging application. It defines two main models: User and Message, handling user information and message storage. Classes: User: Custom user model extending Django's AbstractUser Message: Model for storing messages between users, with support for threaded conversations """ from django.contrib.auth.models import AbstractUser, Group, Permission from django.db import models from django.utils import timezone from datetime import timedelta def get_default_timestamp(): return timezone.now() + timedelta(hours=2) class User(AbstractUser): username = models.CharField(max_length=50, unique=True) first_name = models.CharField(max_length=50) last_name = models.CharField(max_length=50) password = models.CharField(max_length=128) groups = models.ManyToManyField( Group, related_name="custom_user_set", # Custom related_name to avoid clashes blank=True ) user_permissions = models.ManyToManyField( Permission, related_name="custom_user_permissions", # Custom related_name to avoid clashes blank=True ) def __str__(self): return self.username class Message(models.Model): sender = models.ForeignKey(User, on_delete=models.CASCADE, related_name="sent_messages") recipient = models.ForeignKey(User, on_delete=models.CASCADE, related_name="received_messages") content = models.TextField(max_length=1024) #timestamp = models.DateTimeField(auto_now_add=True) timestamp = models.DateTimeField(auto_now_add=False, default=get_default_timestamp) parent_message = models.ForeignKey('self', null=True, blank=True, on_delete=models.CASCADE) def __str__(self): return f"Message from {self.sender} to {self.recipient} at {self.timestamp}"
